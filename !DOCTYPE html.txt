        // --- 1. Background Animations ---
        function createDecorations() {
            const container = document.getElementById('decorations');
            const icons = ['‚ù§Ô∏è', 'üå∏', 'üíñ', 'üåπ', '‚ú®'];
            
            setInterval(() => {
                const element = document.createElement('div');
                element.className = Math.random() > 0.5 ? 'heart' : 'petal';
                element.innerHTML = icons[Math.floor(Math.random() * icons.length)];
                element.style.left = Math.random() * 100 + 'vw';
                element.style.fontSize = (Math.random() * 20 + 10) + 'px';
                element.style.animationDuration = (Math.random() * 3 + 2) + 's';
                
                container.appendChild(element);
                
                setTimeout(() => {
                    element.remove();
                }, 5000);
            }, 300);
        }

        // --- 2. Navigation Logic ---
        function nextScreen(screenId) {
            // Hide all screens
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            // Show target screen
            document.getElementById(screenId).classList.add('active');

            // If it's the final screen, trigger a "heart burst"
            if(screenId === 'screen-yes') {
                triggerBurst();
            }
        }

        // --- 3. Password Check ---
        function checkPassword() {
            const pass = document.getElementById('passInput').value;
            const error = document.getElementById('passError');
            
            if (pass === "25072005") {
                nextScreen('screen-q1');
            } else {
                error.innerText = "Oops‚Ä¶ that‚Äôs not it üëÄ Try again";
                document.getElementById('passInput').value = "";
            }
        }

        // --- 4. Special Effects ---
        function triggerBurst() {
            for(let i=0; i<30; i++) {
                setTimeout(() => {
                    const burst = document.createElement('div');
                    burst.className = 'heart';
                    burst.innerHTML = '‚ù§Ô∏è';
                    burst.style.left = '50vw';

                    burst.style.animation = 'fadeIn 1s forwards';
                    document.body.appendChild(burst);
                }, i * 50);
            }
        }

        // Initialize
